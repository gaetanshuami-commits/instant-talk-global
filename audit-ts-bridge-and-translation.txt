export interface BridgeConnectionState {
  connected: boolean;
  connecting: boolean;
  error: string | null;
  lastMessageTime: number;
}

export interface BridgeMessage {
  type: 'transcript' | 'translation' | 'error' | 'bridge_ready';
  text?: string;
  language?: string;
  message?: string;
}

class BridgeConnection {
  private ws: WebSocket | null = null;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;
  private reconnectDelay = 2000;
  private heartbeatInterval: NodeJS.Timeout | null = null;
  private messageHandlers: Set<(msg: BridgeMessage) => void> = new Set();
  private stateListeners: Set
    (state: BridgeConnectionState) => void
  > = new Set();
  private state: BridgeConnectionState = {
    connected: false,
    connecting: false,
    error: null,
    lastMessageTime: 0,
  };

  connect(): void {
    if (this.state.connecting || this.state.connected) {
      return;
    }

    this.updateState({ connecting: true, error: null });

    try {
      const protocol =
        typeof window !== 'undefined' && window.location.protocol === 'https:'
          ? 'wss:'
          : 'ws:';
      const host =
        typeof window !== 'undefined'
          ? window.location.host
          : 'localhost:3000';
      const url = `${protocol}//${host}/api/deepgram-bridge`;

      this.ws = new WebSocket(url);

      this.ws.onopen = () => {
        console.log('[Bridge] Connected');
        this.reconnectAttempts = 0;
        this.updateState({ connected: true, connecting: false });
        this.startHeartbeat();
      };

      this.ws.onmessage = (event) => {
        try {
          this.state.lastMessageTime = Date.now();
          const data = JSON.parse(event.data) as BridgeMessage;
          this.messageHandlers.forEach((handler) => handler(data));
        } catch (error) {
          console.error('[Bridge] Message parse error:', error);
        }
      };

      this.ws.onerror = (error) => {
        console.error('[Bridge] Error:', error);
        this.updateState({
          connected: false,
          error: 'WebSocket error',
        });
      };

      this.ws.onclose = () => {
        console.log('[Bridge] Disconnected');
        this.stopHeartbeat();
        this.updateState({ connected: false, connecting: false });
        this.attemptReconnect();
      };
    } catch (error) {
      console.error('[Bridge] Connection failed:', error);
      this.updateState({
        connecting: false,
        connected: false,
        error: String(error),
      });
      this.attemptReconnect();
    }
  }

  private attemptReconnect(): void {
    if (this.reconnectAttempts >= this.maxReconnectAttempts) {
      this.updateState({
        error: `Failed to connect after ${this.maxReconnectAttempts} attempts`,
      });
      return;
    }

    this.reconnectAttempts++;
    const delay = Math.min(
      this.reconnectDelay * Math.pow(2, this.reconnectAttempts),
      30000
    );
    console.log(`[Bridge] Reconnecting in ${delay}ms (attempt ${this.reconnectAttempts})`);

    setTimeout(() => this.connect(), delay);
  }

  private startHeartbeat(): void {
    this.heartbeatInterval = setInterval(() => {
      if (this.ws?.readyState === WebSocket.OPEN) {
        this.ws.send(JSON.stringify({ type: 'ping' }));
      }
    }, 30000);
  }

  private stopHeartbeat(): void {
    if (this.heartbeatInterval) {
      clearInterval(this.heartbeatInterval);
      this.heartbeatInterval = null;
    }
  }

  sendAudio(audioData: Int16Array, targetLanguage: string): void {
    if (this.ws?.readyState !== WebSocket.OPEN) {
      console.warn('[Bridge] WebSocket not open, dropping audio frame');
      return;
    }

    try {
      this.ws.send(
        JSON.stringify({
          type: 'audio',
          audio: Array.from(audioData),
          targetLanguage,
        })
      );
    } catch (error) {
      console.error('[Bridge] Failed to send audio:', error);
    }
  }

  stopTranscription(): void {
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify({ type: 'stop' }));
    }
  }

  onMessage(handler: (msg: BridgeMessage) => void): () => void {
    this.messageHandlers.add(handler);
    return () => {
      this.messageHandlers.delete(handler);
    };
  }

  onStateChange(handler: (state: BridgeConnectionState) => void): () => void {
    this.stateListeners.add(handler);
    handler(this.state);
    return () => {
      this.stateListeners.delete(handler);
    };
  }

  private updateState(partial: Partial<BridgeConnectionState>): void {
    this.state = { ...this.state, ...partial };
    this.stateListeners.forEach((listener) => listener(this.state));
  }

  disconnect(): void {
    this.stopHeartbeat();
    if (this.ws) {
      this.ws.close();
      this.ws = null;
    }
    this.updateState({ connected: false, connecting: false });
  }

  getState(): BridgeConnectionState {
    return { ...this.state };
  }
}

  async function playStreamingAudio(res: Response, seq: number, traceId: string) {

      return;

    if (!MediaSourceCtor || !MediaSourceCtor.isTypeSupported || !MediaSourceCtor.isTypeSupported("audio/mpeg")) {

    currentAudioUrlRef.current = objectUrl;

      mediaSource.addEventListener("error", () => reject(new Error("MediaSource error")), { once: true });
    });

    if (seq !== ttsSequenceRef.current) {
      throw new Error("Obsolete TTS sequence");
    }

    const sourceBuffer = mediaSource.addSourceBuffer("audio/mpeg");
    sourceBuffer.mode = "sequence";

    const reader = res.body.getReader();
    const queue: Uint8Array[] = [];
    let appending = false;
    let finished = false;
    let playbackStarted = false;
    let chunkCount = 0;
    let totalBytes = 0;
    let firstChunkAt: number | null = null;

    const waitForIdle = () =>
      new Promise<void>((resolve) => {
        if (!appending && !sourceBuffer.updating) {
          resolve();
          return;
        }

        const done = () => {
          sourceBuffer.removeEventListener("updateend", done);
          resolve();
        };

        sourceBuffer.addEventListener("updateend", done);
      });

    const flushQueue = async () => {
      if (appending || sourceBuffer.updating || queue.length === 0) return;
      appending = true;

      try {
        const chunk = queue.shift();
        if (!chunk) return;
        sourceBuffer.appendBuffer(chunk);

        await new Promise<void>((resolve) => {
          const done = () => {
            sourceBuffer.removeEventListener("updateend", done);
            resolve();
          };
          sourceBuffer.addEventListener("updateend", done);
        });

        if (!playbackStarted) {
          playbackStarted = true;
          await publishTrace({
            traceId,
            stage: "playback_start_streaming",
            t: nowMs(),
            meta: {
              firstChunkAt,
              chunkCount,
              totalBytes,
            },
          });

          try {
            await audio.play();
          } catch {}
        }
      } finally {
        appending = false;

        if (queue.length > 0) {
          await flushQueue();
        } else if (finished && !sourceBuffer.updating && mediaSource.readyState === "open") {
          try {
            mediaSource.endOfStream();
          } catch {}
        }
      }
    };

    audio.onended = async () => {
      if (seq !== ttsSequenceRef.current) return;

      playbackDepthRef.current = Math.max(0, playbackDepthRef.current - 1);

      await publishTrace({
        traceId,
        stage: "playback_end",
        t: nowMs(),
        meta: { chunkCount, totalBytes },
      });

      if (playbackDepthRef.current === 0) {
        resumeSpeechCapture();
        if (unmuteFnRef.current) {
          await unmuteFnRef.current();
        }
      }

      if (currentAudioUrlRef.current) {
        try {
          URL.revokeObjectURL(currentAudioUrlRef.current);
        } catch {}
        currentAudioUrlRef.current = null;
      }
    };

    while (true) {
      const { done, value } = await reader.read();

      if (seq !== ttsSequenceRef.current) {
        try {
          reader.cancel();
        } catch {}
        throw new Error("Obsolete TTS sequence");
      }

      if (done) {
        finished = true;

        await publishTrace({
          traceId,
          stage: "tts_stream_complete",
          t: nowMs(),
          meta: { chunkCount, totalBytes },
        });

        await waitForIdle();

        if (queue.length === 0 && !sourceBuffer.updating && mediaSource.readyState === "open") {
          try {
            mediaSource.endOfStream();
          } catch {}
        }
        break;
      }

      if (value && value.byteLength > 0) {
        chunkCount += 1;
        totalBytes += value.byteLength;

        if (firstChunkAt === null) {
          firstChunkAt = nowMs();
          await publishTrace({
            traceId,
            stage: "tts_first_chunk_received",
            t: firstChunkAt,
            meta: { bytes: value.byteLength },
          });
        }

        queue.push(value);
        await flushQueue();
      }
    }
  }

        }

  async function buildSpeechStream() {

        autoGainControl: true,

        window.sessionStorage.setItem("instant-talk-user-id", userId);

import { NextRequest, NextResponse } from "next/server";
import { GoogleGenerativeAI } from "@google/generative-ai";

export const runtime = "nodejs";

function normalizeLang(lang: unknown) {
  return String(lang ?? "")
    .trim()
    .toUpperCase()
    .replace(/[^A-Z-]/g, "");
}

function nowMs() {
  return Date.now();
}

async function fetchWithTimeout(url: string, init: RequestInit, timeoutMs: number) {
  const controller = new AbortController();
  const timer = setTimeout(() => controller.abort(), timeoutMs);

  try {
    return await fetch(url, {
      ...init,
      signal: controller.signal,
      cache: "no-store",
    });
  } finally {
    clearTimeout(timer);
  }
}

export async function POST(req: NextRequest) {
  const startedAt = nowMs();
  const traceId = req.headers.get("x-trace-id") || crypto.randomUUID();
  const clientSpeechFinalAt = Number(req.headers.get("x-client-speech-final-at") || "0") || null;

  try {
    const { text, targetLang, sourceLang } = await req.json();

    const cleanText = String(text ?? "").trim();
    const normalizedTarget = normalizeLang(targetLang);
    const normalizedSource = normalizeLang(sourceLang);

    if (!cleanText || !normalizedTarget) {
      return NextResponse.json({ error: "Missing parameters" }, { status: 400 });
    }

    if (normalizedSource && normalizedSource === normalizedTarget) {
      return NextResponse.json({
        translatedText: cleanText,
        provider: "bypass",
        detectedSourceLang: normalizedSource,
        traceId,
        timing: {
          startedAt,
          finishedAt: nowMs(),
          durationMs: nowMs() - startedAt,
          clientSpeechFinalAt,
        },
      });
    }

    const deeplKey = process.env.DEEPL_API_KEY;
    const geminiKey = process.env.GEMINI_API_KEY;

    if (deeplKey) {
      try {
        const params = new URLSearchParams();
        params.append("text", cleanText);
        params.append("target_lang", normalizedTarget);

        if (normalizedSource) {
          params.append("source_lang", normalizedSource);
        }

        const providerStartedAt = nowMs();

        const response = await fetchWithTimeout(
          "https://api.deepl.com/v2/translate",
          {
            method: "POST",
            headers: {
              Authorization: `DeepL-Auth-Key ${deeplKey}`,
              "Content-Type": "application/x-www-form-urlencoded",
            },
            body: params.toString(),
          },
          1800
        );

        const raw = await response.text();

        if (response.ok) {
          const data = JSON.parse(raw);
          const finishedAt = nowMs();

          return NextResponse.json({
            translatedText: data?.translations?.[0]?.text ?? cleanText,
            provider: "deepl",
            detectedSourceLang: data?.translations?.[0]?.detected_source_language ?? normalizedSource ?? null,
            traceId,
            timing: {
              startedAt,
              providerStartedAt,
              finishedAt,
              durationMs: finishedAt - startedAt,
              providerDurationMs: finishedAt - providerStartedAt,
              clientSpeechFinalAt,
            },
          });
        }
      } catch (error) {
        console.warn("DeepL fast path failed:", error);
      }
    }

    if (geminiKey) {
      try {
        const genAI = new GoogleGenerativeAI(geminiKey);
        const model = genAI.getGenerativeModel({
          model: "gemini-2.5-flash",
        });

        const prompt =
          `Translate the following spoken sentence into ${normalizedTarget}. ` +
          `Return only the translated sentence. No quotes. No markdown. ` +
          `Keep it natural and concise. Sentence: ${cleanText}`;

        const providerStartedAt = nowMs();

        const result = await Promise.race([
          model.generateContent(prompt),
          new Promise((_, reject) =>
            setTimeout(() => reject(new Error("Gemini timeout")), 2500)
          ),
        ]) as Awaited<ReturnType<typeof model.generateContent>>;

        const translatedText = result.response.text().trim();

        if (translatedText) {
          const finishedAt = nowMs();

          return NextResponse.json({
            translatedText,
            provider: "gemini",
            detectedSourceLang: normalizedSource || null,
            traceId,
            timing: {
              startedAt,
              providerStartedAt,
              finishedAt,
              durationMs: finishedAt - startedAt,
              providerDurationMs: finishedAt - providerStartedAt,
              clientSpeechFinalAt,
            },
          });
        }
      } catch (error) {
        console.warn("Gemini fallback failed:", error);
      }
    }

    return NextResponse.json(
      { error: "No translation provider available", traceId },
      { status: 500 }
    );
  } catch (error) {
    console.error("Translate error:", error);
    return NextResponse.json({ error: "Translation failed", traceId }, { status: 500 });
  }
}

export async function POST(req: Request) {
  const startedAt = nowMs();
  const traceId = req.headers.get("x-trace-id") || crypto.randomUUID();

  try {
    const body = await req.json();
    const { text, voiceId } = body;

    const apiKey = process.env.ELEVENLABS_API_KEY;

    if (!apiKey) {
      return NextResponse.json({ error: "Cle ElevenLabs manquante", traceId }, { status: 500 });
    }

    const cleanText = String(text ?? "").trim();
    if (!cleanText) {
      return NextResponse.json({ error: "Texte manquant", traceId }, { status: 400 });
    }

    const resolvedVoiceId = String(voiceId || DEFAULT_VOICE_ID).trim();

    const providerStartedAt = nowMs();

    const response = await fetchWithTimeout(
      `https://api.elevenlabs.io/v1/text-to-speech/${resolvedVoiceId}/stream?optimize_streaming_latency=3&output_format=mp3_44100_128`,
      {
        method: "POST",
        headers: {
          "xi-api-key": apiKey,
          "Content-Type": "application/json",
          "Accept": "audio/mpeg",
        },
        body: JSON.stringify({
          text: cleanText,
          model_id: DEFAULT_MODEL_ID,
          voice_settings: {
            stability: 0.3,
            similarity_boost: 0.7,
          },
        }),
      },
      15000
    );

    const firstByteAt = nowMs();

    if (!response.ok) {
      const errText = await response.text().catch(() => "");
      console.error("ElevenLabs streaming error:", response.status, errText);
      return NextResponse.json(
        {
          error: `Erreur ElevenLabs: ${response.status}`,
          traceId,
          timing: {
            startedAt,
            providerStartedAt,
            firstByteAt,
            durationMs: firstByteAt - startedAt,
            providerDurationMs: firstByteAt - providerStartedAt,
          },
        },
        { status: 500 }
      );
    }

    return new NextResponse(response.body, {
      headers: {
        "Content-Type": "audio/mpeg",
        "Transfer-Encoding": "chunked",
        "Cache-Control": "no-store, no-cache, must-revalidate, proxy-revalidate",
        "Pragma": "no-cache",
        "Expires": "0",
        "x-trace-id": traceId,
        "x-tts-started-at": String(startedAt),
        "x-tts-provider-started-at": String(providerStartedAt),
        "x-tts-first-byte-at": String(firstByteAt),
      },
    });
  } catch (error) {
    console.error("TTS route error:", error);
    return NextResponse.json({ error: "Erreur TTS", traceId }, { status: 500 });
  }
}
